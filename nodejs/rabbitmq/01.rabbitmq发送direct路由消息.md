var amqp = require("amqp");

var connection = amqp.createConnection({
    'host': '192.168.99.234',
    'port': 5672,
    'login': 'hitrader',
    'password': 'hitrader123',
    'vhost': '/',
});

var exchangeName = "ExAutoToCpp.Dir";
var routingKey = "AutobindToCpp";
var exchangeType = {'type': 'direct', 'durable': true, 'autoDelete': false, 'confirm': false};

connection.on('ready', function(){
    console.log("connection success");
    connection.exchange(exchangeName, exchangeType, function(exchange){
        console.log("start publish");
        if(exchange.publish(routingKey, '{"aaa": 123, "bbb": 456}', "")){
            console.log("publish success");
            connection.end();
            connection.destroy();
        }else{
            console.log("publish failed");
            connection.end();
            connection.destroy();
        }
    });
});
